1)Запись данных и хранение данных

3)Обеспечение конфиденциальности данных. (т.е. нужен ключ, чтобы смотреть данные)

//Этот класс отедльно от main файла, т.е. в отдельном файле
namespace MyApp 
{
    public class Student 
    {
        //Члены класса - обязательно с маленькой буквой поля, а Методы с большой; в JavaScript с маленькой
        protected int age; //Поля закрыты
        protected string name; //Инкапсуляция

        public Student(string _name, int _age) {
            this.age = _age;
            this.name = _name;
        }

        //mutate - запись
        public void ChangeAge(int _age) {
            this.age = _age;
        }

        //access - чтенеи 
        public int ReadAge() {
            return this.age;
        }

        //Property для имени
        public string Name
        {
            get { 
                return this.name;
            }

            set { 
                this.name = value;
            }
        }

        public void Info() {
            Console.WriteLine("Ваше имя: {0}, Вам столько {1} лет", name, age);
        }
    }
}
